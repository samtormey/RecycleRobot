\begin{block}{\centering Introduction}
Improving the efficiency of recycling through automation was the inspiration for our project.    \\

\centering \includegraphics[height=10cm, width = 12cm]{figures/mrf-recycling-system.jpg}\\

We model the removal of objects from a conveyor belt with a MATLAB simulation of a SCARA robot.\\

We solve a constrained optimization problem to find optimal (minimum-time) path between starting and
ending robot configurations. \\
 
\end{block}

\begin{block}{\centering Stochastic Elliptic Problem} 

\begin{align*}
-\nabla\cdot(a_N(\bxi,\bx)\nabla u(\bxi,\bx)) &= f(\bx) \;\; &&\text{in} \; \cD\times\Gamma\\
u(\bxi,\bx) &= 0 \;\; &&\text{on} \; \partial\cD\times\Gamma
\end{align*}
\noindent
\hspace{1em}where\\
\vspace{0.5em}
\hspace{2em} $u(\bxi,\bx)$ - a state variable (e.g. pressure)\\
\hspace{2em} $a_N(\bxi,\bx)$ - a random diffusion coefficient (e.g. permeability) \\
\hspace{6.2em} given by
%\vspace{1em}
%Under the finite noise assumption $a(\bxi,\bx)\approx a_N(\bxi,\bx)$ with\\
\begin{align*}
\log{(a_N(\bxi,\bx))} = \sum\limits_{n=1}^N Y_n(\bxi)\zeta_n\psi_n(\bx)
\end{align*}

\begin{block}{\centering Objective}

Estimate uncertainty in $Q(u(\bxi,\bx))$ via
\begin{align*}
\centering \text{Mean value: } \;\; &\E[Q(u(\cdot,\bx))] \\
\centering \text{Semi-deviation: } \;\; &\E[\max{\{(Q(u(\cdot,\bx)) - \E[Q(u(\cdot,\bx))]),0\}}]
\end{align*}

%Our goal is to estimate the uncertainty in some quantity of interest $Q(u(\bxi,\bx))$ that depends on the solution $u(\bxi,\bx)$ such as\\
%\vspace{0.5em}
%\begin{align*}
%Q(u(\bxi,\bx)) = u(x^*)
%\end{align*}
% \hspace{2em} Mean value $\E[Q(u(\cdot,\bx))]$\\
% \hspace{2em} Semideviation $\E[\max{\{(Q(u(\cdot,\bx)) - \E[Q(u(\cdot,\bx))]),0\}}]$
\vspace{0.5em}



Challenges:
\begin{itemize}
\item Integration over high-dimensional parameter space $\Gamma$;
\item Each evaluation of quantity of interest requires solving a PDE (cost);
\item The integrand is non-smooth (max function).
\end{itemize}

\vspace{0.5em}
Need non-intrusive robust and efficient sampling methods that take into account local behavior of the integrand (quantity of interest).
\end{block}

\begin{block}{\centering Hierarchical Basis}

\vspace{1em}

\begin{columns}[T]
\begin{column}{22cm}{}
Represent $f$ using hierarchical basis functions (with varying support):\\
\begin{equation*}
f(\bxi) \approx \sum\limits_{\bl} f_{\bl}(\bxi), \;\;\; f_{\bl}(\bxi) = \sum\limits_{\bi\;\;\text{odd}}  s_{\bl,\bi} \phi_{\bl,\bi}(\bxi)
\end{equation*}
%\noindent
%where coefficients $s_{\bl,\bi}$ are determined from 
%\begin{equation*}
%s_{\bl,\bi} = ( \prod_{j=1}^d [\; -\frac{1}{2} \;\; 1 \;\; -\frac{1}{2} \;]_{\xi_{l_j,i_j},l_j} ) f
%\end{equation*}
\noindent
where $d$-dimensional functions $\phi_{\bl,\bi}(\bxi)$ are defined as
\begin{equation*}
\phi_{\bl,\bi}(\bxi) = \prod_{j=1}^d \phi_{l_j,i_j}(\xi_j)
\end{equation*}
Coefficients $s_{\bl,\bi}$ are called \textit{hierarchical surpluses} as they represent hierarchichal increments between neighboring levels.
\end{column}
\begin{column}{12cm}{}
\centering\includegraphics[height=16cm, width = 10cm]{figures/hierbasis_v2} \\
\vspace{1em}
\centering \scriptsize{1D hierarchical piecewise linear basis}
\end{column}

%Scheme of hierarchical subspaces in 2D
%\end{column}
\end{columns}


\end{block}

 \begin{block}{\centering Adaptivity (hASG)}

Hierarchical surpluses $s_{\bl,\bi}$ are a natural local error indicator.\\

Thus, refining only points with relatively large surpluses we obtain an adaptive grid (with more points spent in non-smooth areas).\\
%
%The idea of adaptivity is to refine only those points $\bxi_{\bl,\bi}$ for which $|s_{\bl,\bi}|>\epsilon$ (where $\epsilon$ is prescribed tolerance).\\
